% --- 定义报错宏：如果字体缺失，则停止编译并显示具体错误信息 ---
\newcommand{\FontMissingError}[2]{
  \GenericError{(Font Error)}{^^J
    ============================================================^^J
    错误: 找不到字体 "#1" !^^J
    该字体是论文格式必需的。请采取以下措施之一：^^J
    1. 在系统中安装该字体 (Windows/macOS 系统字体库);^^J
    2. 将字体文件 "#2" 放入项目根目录下的 [字体] 文件夹中。^^J
    ============================================================}{}{}%
}
% --- 辅助宏：自动选择 1.系统字体 -> 2.本地字体 -> 3.报错 ---
% 参数：#1 内部族名, #2 系统字体名, #3 本地文件名
\newcommand{\DeployCJKFont}[3]{
    \IfFontExistsTF{#2}{
        \setCJKfamilyfont{#1}{#2}
    }{
        \IfFileExists{./字体/#3}{
            \setCJKfamilyfont{#1}{#3}[Path=./字体/]
        }{
            \FontMissingError{#2}{#3}
        }
    }
}
% --- 1. 设置英文字体 (Times New Roman) ---
\IfFontExistsTF{Times New Roman}{
    \setmainfont{Times New Roman}
}{
    \IfFileExists{./字体/TIMES.TTF}{
        \setmainfont{TIMES.TTF}[
            Path=./字体/,
            BoldFont=TIMESBD.TTF,
            ItalicFont=TIMESI.TTF,
            BoldItalicFont=TIMESBI.TTF
        ]
    }{
        \FontMissingError{Times New Roman}{TIMES.TTF / TIMESBD.TTF 等}
    }
}
